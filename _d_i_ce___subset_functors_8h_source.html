115  ZNSSD_Gamma_Functor(intensity_device_view_1d ref_intensities,
116  intensity_device_view_1d def_intensities,
117  bool_device_view_1d is_active,
118  bool_device_view_1d is_deactivated_this_step,
119  const scalar_t & mean_r,
120  const scalar_t & mean_d,
121  const scalar_t & mean_sum_r,
122  const scalar_t & mean_sum_d):
123  ref_intensities_(ref_intensities),
124  def_intensities_(def_intensities),
125  is_active_(is_active),
126  is_deactivated_this_step_(is_deactivated_this_step),
127  mean_r_(mean_r),
128  mean_d_(mean_d),
129  mean_sum_r_(mean_sum_r),
130  mean_sum_d_(mean_sum_d){}
132  KOKKOS_INLINE_FUNCTION
133  void operator()(const int_t pixel_index, scalar_t & gamma) const{
134  if(is_active_(pixel_index)&!is_deactivated_this_step_(pixel_index)){
135  scalar_t value = (def_intensities_(pixel_index)-mean_d_)/mean_sum_d_ - (ref_intensities_(pixel_index)-mean_r_)/mean_sum_r_;
136  //printf("%i value: %e %e %e %e %e %e %e\n",pixel_index,value,def_intensities_(pixel_index),mean_d_,mean_sum_d_,ref_intensities_(pixel_index),mean_r_,mean_sum_r_);
137  gamma += value*value;
138  }
139  }
140 };
