59 template <typename T>
60 struct Point_Cloud
61 {
63  struct Point
64  {
65  T x,y,z;
66  };
68  std::vector<Point> pts;
70  inline size_t kdtree_get_point_count() const { return pts.size(); }
72  inline T kdtree_distance(const T *p1, const size_t idx_p2,size_t /*size*/) const
73  {
74  const T d0=p1[0]-pts[idx_p2].x;
75  const T d1=p1[1]-pts[idx_p2].y;
76  const T d2=p1[2]-pts[idx_p2].z;
77  return d0*d0+d1*d1+d2*d2;
78  }
82  inline T kdtree_get_pt(const size_t idx, int dim) const
83  {
84  if (dim==0) return pts[idx].x;
85  else if(dim==1) return pts[idx].y;
86  else return pts[idx].z;
87  }
91  template <class BBOX>
92  bool kdtree_get_bbox(BBOX& /*bb*/) const { return false; }
93 };
