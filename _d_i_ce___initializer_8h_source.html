59 template <typename T>
60 struct Point_Cloud
61 {
63  struct Point
64  {
66  T x;
68  T y;
70  T z;
71  };
73  std::vector<Point> pts;
75  inline size_t kdtree_get_point_count() const { return pts.size(); }
77  inline T kdtree_distance(const T *p1, const size_t idx_p2,size_t /*size*/) const
78  {
79  const T d0=p1[0]-pts[idx_p2].x;
80  const T d1=p1[1]-pts[idx_p2].y;
81  const T d2=p1[2]-pts[idx_p2].z;
82  return d0*d0+d1*d1+d2*d2;
83  }
87  inline T kdtree_get_pt(const size_t idx, int dim) const
88  {
89  if (dim==0) return pts[idx].x;
90  else if(dim==1) return pts[idx].y;
91  else return pts[idx].z;
92  }
96  template <class BBOX>
97  bool kdtree_get_bbox(BBOX& /*bb*/) const { return false; }
98 };
