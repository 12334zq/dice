871  void sync_fields_dist_to_all(){
872  if(target_field_descriptor_==ALL_OWNED) return; // NO-OP
873 #if DICE_MPI
874  if(target_field_descriptor_==DISTRIBUTED){
875  fields_->do_import(*dist_fields_,*importer_);
876  fields_nm1_->do_import(*dist_fields_nm1_,*importer_);
877  }
878  else if(target_field_descriptor_==DISTRIBUTED_GROUPED_BY_SEED){
879  fields_->do_import(*seed_dist_fields_,*seed_importer_);
880  fields_nm1_->do_import(*seed_dist_fields_,*seed_importer_);
881  }
882  else{
883  assert(false && "Error: unknown field descriptor.");
884  }
885  distributed_fields_being_modified_ = false;
886 #else
887  return; // NO-OP without MPI
888 #endif
889  }

892  void sync_fields_all_to_dist(){
893  if(target_field_descriptor_==ALL_OWNED) return; // NO-OP
894 #if DICE_MPI
895  distributed_fields_being_modified_ = true;
896  if(target_field_descriptor_==DISTRIBUTED){
897  dist_fields_->do_export(*fields_,*exporter_);
898  dist_fields_nm1_->do_export(*fields_nm1_,*exporter_);
899  }
900  else if(target_field_descriptor_==DISTRIBUTED_GROUPED_BY_SEED){
901  seed_dist_fields_->do_export(*fields_,*seed_exporter_);
902  seed_dist_fields_nm1_->do_export(*fields_nm1_,*seed_exporter_);
903  }
904  else{
905  assert(false && "Error: unknown field descriptor.");
906  }
907  //Teuchos::RCP<Teuchos::FancyOStream> fos = Teuchos::fancyOStream(Teuchos::rcpFromRef(std::cout));
908  //dist_fields_->describe(*fos,Teuchos::VERB_EXTREME);
909 #else
910  return; // NO-OP without MPI
911 #endif
