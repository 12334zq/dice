842  void sync_fields_dist_to_all(){
843  if(target_field_descriptor_==ALL_OWNED) return; // NO-OP
844 #if DICE_MPI
845  if(target_field_descriptor_==DISTRIBUTED){
846  fields_->do_import(*dist_fields_,*importer_);
847  fields_nm1_->do_import(*dist_fields_nm1_,*importer_);
848  }
849  else if(target_field_descriptor_==DISTRIBUTED_GROUPED_BY_SEED){
850  fields_->do_import(*seed_dist_fields_,*seed_importer_);
851  fields_nm1_->do_import(*seed_dist_fields_,*seed_importer_);
852  }
853  else{
854  assert(false && "Error: unknown field descriptor.");
855  }
856  distributed_fields_being_modified_ = false;
857 #else
858  return; // NO-OP without MPI
859 #endif
860  }
861 
863  void sync_fields_all_to_dist(){
864  if(target_field_descriptor_==ALL_OWNED) return; // NO-OP
865 #if DICE_MPI
866  distributed_fields_being_modified_ = true;
867  if(target_field_descriptor_==DISTRIBUTED){
868  dist_fields_->do_export(*fields_,*exporter_);
869  dist_fields_nm1_->do_export(*fields_nm1_,*exporter_);
870  }
871  else if(target_field_descriptor_==DISTRIBUTED_GROUPED_BY_SEED){
872  seed_dist_fields_->do_export(*fields_,*seed_exporter_);
873  seed_dist_fields_nm1_->do_export(*fields_nm1_,*seed_exporter_);
874  }
875  else{
876  assert(false && "Error: unknown field descriptor.");
877  }
878  //Teuchos::RCP<Teuchos::FancyOStream> fos = Teuchos::fancyOStream(Teuchos::rcpFromRef(std::cout));
879  //dist_fields_->describe(*fos,Teuchos::VERB_EXTREME);
