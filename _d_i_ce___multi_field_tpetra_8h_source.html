415  scalar_t norm(Teuchos::RCP<MultiField> multifield){
416  TEUCHOS_TEST_FOR_EXCEPTION(this->get_map()->get_num_local_elements()!=
417  multifield->get_map()->get_num_local_elements(),std::runtime_error,
418  "Error, incompatible multifield maps");
419  scalar_t norm = 0.0;
420  for(int_t i=0;i<get_map()->get_num_local_elements();++i){
421  norm += (this->local_value(i)-multifield->local_value(i))*
422  (this->local_value(i)-multifield->local_value(i));
423  }
424  norm = std::sqrt(norm);
425  return norm;
426  }
