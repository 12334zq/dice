401  scalar_t norm(Teuchos::RCP<MultiField> multifield){
402  TEUCHOS_TEST_FOR_EXCEPTION(this->get_map()->get_num_local_elements()!=
403  multifield->get_map()->get_num_local_elements(),std::runtime_error,
404  "Error, incompatible multifield maps");
405  scalar_t norm = 0.0;
406  for(int_t i=0;i<get_map()->get_num_local_elements();++i){
407  norm += (this->local_value(i)-multifield->local_value(i))*
408  (this->local_value(i)-multifield->local_value(i));
409  }
410  norm = std::sqrt(norm);
411  return norm;
412  }
